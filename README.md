To run this application Create .env file with below content

DEBUG=False
SECRET_KEY=your_secret_key_here
DB_NAME=
DB_USER=
DB_PASSWORD=
DB_HOST=your_postgres_aws_db
ALLOWED_HOSTS=your_ec2_host
AWS_ACCESS_KEY_ID=
AWS_SECRET_ACCESS_KEY=
AWS_STORAGE_BUCKET_NAME=your_s3_bucket_name
AWS_S3_REGION_NAME=us-east-1
